主なproxyの設定場所について
●yum
・/etc/yum.conf
# Proxy Setting
proxy=http://proxy.com:portnum
proxy_username=username
proxy_password=password
●wget
・~/.wgetrc
# Proxy Setting
http_proxy=http://proxy.com:portnum
proxy_user=username
proxy_password=password
・/etc/wgetrc
http_proxy = http://username:password@proxy.com:portnum
●git
・~/.gitconfig
[http]
    proxy = http://username:password@proxy.com:portnum
[https]
    proxy = http://username:password@proxy.com:portnum
●curl
・~/.curlrc
proxy-user = "username:password"
proxy = "http://proxy.com:portnum"
●mercurial
・~/.hgrc
[http_proxy]
host=proxy.com:portnum
user=username
passwd=password
●postfix
・/etc/postfix/sasl_passwd
[smtp.gmail.com]:587 username@systena.co.jp:password

■ luaの入ったVIMをインストール
cd /usr/local/src
sudo yum install mercurial ncurses-devel lua lua-devel

vimをmakeインストールする

cd /usr/local/bin/
wget ftp://ftp.vim.org/pub/vim/unix/vim-7.4.tar.bz2
tar xvf vim-7.4.tar.bz2
cd vim74
./configure \
    --with-features=huge \
    --enable-multibyte \
    --enable-luainterp=dynamic \
    --enable-gpm \
    --enable-cscope \
    --enable-fontset
make
make install
